// this program will generate a Hex board and let you play a game 
// the hexboard class will generate a 11x11 board and keep track of the moves
// its uses a graph representation and a spanning tree to check if a player has won
// i made it so the hexboard class can make a board of any size, but for the purpose 
// of the assignment i used a standard 11x11 board
// for the purpose of testing you may want to use a smaller board
// to do that just change the line HexBoard board; to Hexboard board(7); for example


#include <iostream>
#include <string>
#include <time.h>


#include "Node.h"
#include "HexBoard.h"
#include "Playerturn.h"
#include "Player.h"


int main()
{
	// declare the variables
	HexBoard board(11);
	char answer;
	ComputerPlayer human;
	ComputerPlayer computer;
	srand(time(0));

	// print a welcome message and prints the board
	cout << "Welsome to the Game of Hex" << endl;
	cout << "Do you want to go First? (player X): ";
	cin >> answer;
	if (toupper(answer) == 'Y')
	{
		human.SetPlayer('X');
		computer.SetPlayer('O');
	}
	else
	{
		human.SetPlayer('O');
		computer.SetPlayer('X');
	}

	board.printBoard();

	// loops till there is a winner
	while (board.Winner() == 0)
	{
		if (board.turn == Playerturn::X)
		{
			if (human.getStone() == 'X')
			{
				human.Move(board);
			}
			else
			{
				computer.Move(board);
			}
		}
		else
		{
			if (human.getStone() == 'O')
			{
				human.Move(board);
			}
			else
			{
				computer.Move(board);
			}
		}
		board.printBoard();
	}

	// checks which player has won, and prints a message
	if (board.Winner() == 1)
	{
		cout << "Player X Wins!!" << endl;
	}
	else
		cout << "Player O Wins!!" << endl;
	cout << endl;

	return 0;

}
